<!--HTML5 <tag> with respective id-->
<b id="words">Story Drafts</b>

<!--JS script for text change effect-->
<script type="text/javascript">
      (function () {
        var words = ["Story Drafts","Pre Writings","Novel Ideas","Correction Lounges"],
        i = 0;
        setInterval(function(){ $('#words').fadeOut(function(){
            $(this).html(words[(i = (i + 1) % words.length)]).fadeIn();
          }); }, 3000)
      })();
</script>
